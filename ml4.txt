ūüíĽ Python Code (Naive Bayes using sklearn)
from sklearn.naive_bayes import GaussianNB

# Sample data
X = [[1, 20], [2, 21], [3, 22], [10, 30]]
y = [0, 0, 0, 1]

model = GaussianNB()
model.fit(X, y)

# Prediction
print(model.predict([[2, 21]]))
